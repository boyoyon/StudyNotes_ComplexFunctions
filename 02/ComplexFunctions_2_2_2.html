2.2 Cauchy-Riemann の方程式

\(f(z),g(z)\) が \(z_0\) で微分可能とする。

●\(f(z)+g(z)\) の微分係数:　\(f^\prime(z_0)+g^\prime(z_0)\)

(証明) \[ \begin{align} &\lim\limits_{z\rightarrow z_0}\frac{(f(z)+g(z))-(f(z_0)+g(z_0))}{z-z_0}=\lim\limits_{z\rightarrow z_0}\left\{\frac{f(z)-f(z_0)}{z-z_0}+\frac{g(z)-g(z_0)}{z-z_0}\right\} \\ \\ &=\lim\limits_{z\rightarrow z_0}\frac{f(z)-f(z_0)}{z-z_0}+\lim\limits_{z\rightarrow z_0}\frac{g(z)-g(z_0)}{z-z_0}=f^\prime(z_0)+g^\prime(z_0) \end{align} \]


●\(kf(z)\;(kは定数)\) の微分係数:　\(kf^\prime(z_0)\)

(証明) \[ \lim\limits_{z\rightarrow z_0}\frac{kf(z)-kf(z_0)}{z-z_0}=\lim\limits_{z\rightarrow z_0}\frac{k(f(z)-f(z_0))}{z-z_0}=k\lim\limits_{z\rightarrow z_0}\frac{f(z)-f(z_0)}{z-z_0}=kf^\prime(z_0) \]


●\(f(z)g(z)\) の微分係数:　\(f^\prime(z_0)g(z_0)+f(z_0)g^\prime(z_0)\)

(証明) \[ \begin{align} &\lim\limits_{z\rightarrow z_0}\frac{f(z)g(z)-f(z_0)g(z_0)}{z-z_0} =\lim\limits_{z\rightarrow z_0}\frac{f(z)g(z)\overbrace{-f(z_0)g(z)+f(z_0)g(z)}^{=0}-f(z_0)g(z_0)}{z-z_0}\\ \\ &=\lim\limits_{z\rightarrow z_0}\left\{\frac{(f(z)-f(z_0))g(z)}{z-z_0}+\frac{f(z_0)(g(z)-g(z_0))}{z-z_0}\right\}\\ \\ &=\lim\limits_{z\rightarrow z_0}\frac{f(z)-f(z_0)}{z-z_0}\lim\limits_{z\rightarrow z_0}g(z)+f(z_0)\lim\limits_{z\rightarrow z_0}\frac{g(z)-g(z_0)}{z-z_0} \\ \\ &=f^\prime(z_0)g(z_0)+f(z_0)g^\prime(z_0) \end{align} \]


●\(\displaystyle\frac{f(z)}{g(z)}\;(g(z)\neq 0)\) の微分係数:　\(\displaystyle\frac{f^\prime(z_0)g(z_0)-f(z_0)g^\prime(z_0)}{\{g(z_0)\}^2}\)

(証明)

\[ \begin{align} &\lim\limits_{z\rightarrow z_0}\frac{\displaystyle\frac{f(z)}{g(z)}-\displaystyle\frac{f(z_0)}{g(z_0)}}{z-z_0}=\lim\limits_{z\rightarrow z_0}\frac{\displaystyle\frac{f(z)g(z_0)-f(z_0)g(z)}{g(z)g(z_0)}}{z-z_0} \\ \\ &=\lim\limits_{z\rightarrow z_0}\frac{\displaystyle\frac{f(z)g(z_0)\overbrace{-f(z_0)g(z_0)+f(z_0)g(z_0)}^{=0}-f(z_0)g(z)}{g(z)g(z_0)}}{z-z_0} \\ \\ &=\lim\limits_{z\rightarrow z_0}\frac{(f(z)-f(z_0))g(z_0)\displaystyle\frac{1}{g(z)g(z_0)}-(g(z)-g(z_0))f(z_0)\displaystyle\frac{1}{g(z)g(z_0)}}{z-z_0} \\ \\ &=\lim\limits_{z\rightarrow z_0}\frac{(f(z)-f(z_0))g(z_0)-(g(z)-g(z_0))f(z_0)}{z-z_0}\cdot\frac{1}{g(z)g(z_0)} \\ \\ &=\left\{\lim\limits_{z\rightarrow z_0}\frac{f(z)-f(z_0)}{z-z_0}g(z_0) -\lim\limits_{z\rightarrow z_0}\frac{g(z)-g(z_0)}{z-z_0}f(z_0)\right\} \lim\limits_{z\rightarrow z_0}\frac{1}{g(z)g(z_0)} \\ \\ &=\frac{f^\prime(z_0)g(z_0)-f(z_0)g^\prime(z_0)}{\{g(z_0)\}^2} \end{align} \]
